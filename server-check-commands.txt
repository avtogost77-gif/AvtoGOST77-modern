# Команды для проверки на сервере АвтоГОСТ

# 1. Посмотреть что есть в корневой директории сайта
ls -la

# 2. Проверить размер папок
du -sh *

# 3. Найти все HTML файлы
find . -name "*.html" -type f | wc -l

# 4. Проверить права доступа на важные файлы
ls -la .htaccess
ls -la robots.txt
ls -la sitemap.xml

# 5. Проверить что нет лишних файлов
ls -la | grep -E "(\.zip|\.tar|\.gz|\.bak|\.old|\.swp|~$)"

# 6. Проверить структуру директорий
tree -d -L 2

# 7. Проверить последние измененные файлы
find . -type f -mtime -1 -ls

# 8. ПЕРЕД ДЕПЛОЕМ - сделать бекап текущей версии
tar -czf backup-$(date +%Y%m%d-%H%M%S).tar.gz .

# 9. Проверить размер всего сайта
du -sh .

# 10. Проверить что все новые страницы на месте
ls -la | grep -E "(transportnaya|sbornye|marketpleysy|rc-dostavka|gruzoperevozki)"

# ДЕПЛОЙ:
# 1. Распаковать архив: unzip avtogost-deploy-*.zip
# 2. Скопировать файлы: cp -r avtogost-deploy-*/* /путь/к/сайту/
# 3. Проверить права: chmod 644 *.html
# 4. Проверить .htaccess: chmod 644 .htaccess

# ПОСЛЕ ДЕПЛОЯ проверить:
# - Главная страница работает
# - .htaccess правила работают (редиректы)
# - robots.txt доступен: curl https://avtogost77.ru/robots.txt
# - sitemap.xml доступен: curl https://avtogost77.ru/sitemap.xml
# - Новые страницы открываются без .html в URL
