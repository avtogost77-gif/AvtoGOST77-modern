# КОМАНДЫ ДЛЯ ЧИСТОГО ДЕПЛОЯ НА СЕРВЕР

# 1. Создать финальный бэкап перед деплоем
cd /www/wwwroot/avtogost77.ru/
tar -czf backup-final-before-clean-$(date +%Y%m%d-%H%M%S).tar.gz .

# 2. Скачать чистый архив (замени ссылку когда запушим)
wget https://raw.githubusercontent.com/avtogost77-gif/AvtoGOST77-modern/main/avtogost-clean-deploy-20250806-182552.zip

# 3. Удалить весь автогенерированный мусор
rm -rf routes/
rm -rf industries/
rm -rf calculators/
rm -rf blog/gruzoporevozki-moskva-spb/

# 4. Распаковать чистый архив
unzip -o avtogost-clean-deploy-20250806-182552.zip

# 5. Копировать файлы из архива
cp -r avtogost-clean-deploy-20250806-182552/* .

# 6. Очистить
rm -rf avtogost-clean-deploy-20250806-182552/
rm avtogost-clean-deploy-20250806-182552.zip

# 7. Проверить результат
echo "===== ПРОВЕРКА ЧИСТОГО ДЕПЛОЯ ====="
echo "Всего HTML файлов:"
find . -name "*.html" -type f | wc -l
echo ""
echo "Проверка отсутствия автогена:"
ls -la routes/ 2>/dev/null || echo "✅ routes удален"
ls -la industries/ 2>/dev/null || echo "✅ industries удален"
ls -la calculators/ 2>/dev/null || echo "✅ calculators удален"
echo ""
echo "URL в sitemap:"
grep -c "<url>" sitemap.xml
echo ""
echo "✅ Деплой завершен! Только качественный контент!"
